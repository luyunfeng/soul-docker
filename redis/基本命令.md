# 关闭集群
make down
docker-compose stop

# 重新打包
make rebuild
docker-compose build --no-cache

# 进入机器
make bash
docker-compose exec redis-cluster /bin/bash
# 建立连接 默认是 7000 这台
make cli
docker-compose exec redis-cluster /redis/src/redis-cli -p 7000

# 指定机器
docker-compose exec redis-cluster /redis/src/redis-cli -p 7005 -c
